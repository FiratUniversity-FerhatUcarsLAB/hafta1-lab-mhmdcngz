MÜKEMMEL OMLET YAPMA ALGORİTMASI

ALGORITMA MukemmelOmletYap

    // --- 1. Başlangıç Hazırlıkları ---
    BAŞLA
    malzemeler = ["yumurta", "tuz", "karabiber", "tereyağı", "süt veya krema"]
    ekipman = ["tava", "çatal veya çırpıcı", "kase", "spatula"]

    YAZDIR "Omlet yapımına başlıyoruz..."

    // --- 2. Malzeme Kontrolü ---
    HER malzeme İÇİN malzemeler:
        EĞER malzeme mevcut değilse
            YAZDIR malzeme + " eksik, temin edilmesi gerekiyor."
            ÇIKIŞ YAP
        SON
    SON

    // --- 3. Yumurta Kontrolü ---
    yumurtaSayisi ← 3
    YAZDIR yumurtaSayisi + " yumurta seçildi."

    HER yumurta İÇİN yumurtaSayisi:
        EĞER yumurta taze değilse
            YAZDIR "Bayat yumurta tespit edildi. Yeni yumurta seçiliyor..."
            yumurta değiştir
        SON
    SON

    // --- 4. Karışımı Hazırlama ---
    YAZDIR "Yumurtalar kırılıyor..."
    karisim ← yumurtalarıKır() + tuz + karabiber + az miktarda süt

    YAZDIR "Malzemeler karıştırılıyor..."
    karıştırmaSüresi ← 0

    WHILE karıştırmaSüresi < 30 saniye:
        karıştır()
        karıştırmaSüresi ← karıştırmaSüresi + 5
    SON

    YAZDIR "Karışım hazır!"

    // --- 5. Tava Hazırlığı ---
    YAZDIR "Tava ısıtılıyor..."
    tavaSıcaklık ← 0

    WHILE tavaSıcaklık < 180 derece:
        tavaSıcaklık ← tavaSıcaklık + 30
        BEKLE 2 saniye
    SON

    YAZDIR "Tava yeterince sıcak!"
    tava ← tereyağıEkle(tava)

    // --- 6. Pişirme Aşaması ---
    YAZDIR "Omlet karışımı tavaya dökülüyor..."
    tavayaDök(karisim)

    pişmeSeviyesi ← 0

    WHILE pişmeSeviyesi < 100:
        EĞER altKısmıKızardı() ise
            omletiKarıştır()
        SON
        BEKLE 5 saniye
        pişmeSeviyesi ← pişmeSeviyesi + 20
    SON

    // --- 7. Son Dokunuş ---
    EĞER omletAltinRengiMi() ise
        YAZDIR "Omlet mükemmel bir renkte!"
    DEĞİLSE
        YAZDIR "Biraz fazla pişmiş, ama yenebilir."
    SON

    // --- 8. Servis ---
    YAZDIR "Omlet tabağa alınıyor..."
    tabağaKoy(omlet)
    YAZDIR "Afiyet olsun! 🍳"

    BİTİR
SON ALGORITMA
