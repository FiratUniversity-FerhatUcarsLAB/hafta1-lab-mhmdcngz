TELEFON REHBERİNİ ALFABETİK DÜZENLEME

Algoritma: TelefonRehberiniAlfabetikSirala

// Girdi: Rehber - Kişi kayıtlarından oluşan bir dizi/liste (her kayıtta: isim, soyisim, telefon)
// Çıktı: Alfabetik olarak sıralanmış rehber

Başla

    // 1. Rehberin boş olup olmadığını kontrol et
    Eğer Rehber'in uzunluğu = 0 ise
        Yaz("Rehberde kayıt yok, sıralama yapılmayacak.")
        Bitir

    // 2. Sıralama algoritması başlat (Bubble Sort örneğiyle)
    n ← Rehber'in uzunluğu

    // Dış döngü: Listenin başından sonuna kadar tekrar et
    i ← 0
    Tekrar i < n-1 olduğu sürece
        // İç döngü: Her seferinde dizinin sonuna kadar ilerle, karşılaştırma yap
        j ← 0
        Tekrar j < n-i-1 olduğu sürece

            // 3. Karşılaştırılacak iki kayıt seç
            kisi1 ← Rehber[j]
            kisi2 ← Rehber[j+1]

            // 4. Kayıtların isimlerini karşılaştır
            // Not: Büyük/küçük harf farkını kaldırmak için hepsini küçük harfe çeviriyoruz
            isim1 ← KüçükHarfeCevir(kisi1.isim)
            isim2 ← KüçükHarfeCevir(kisi2.isim)

            // 5. Eğer birinci isim ikinci isimden sonra geliyorsa, yer değiştir
            Eğer isim1 > isim2 ise
                // Yer değiştir
                temp ← Rehber[j]
                Rehber[j] ← Rehber[j+1]
                Rehber[j+1] ← temp

            // 6. Eğer isimler aynıysa, soyisimleri karşılaştır
            Değilse Eğer isim1 = isim2 ise
                soyisim1 ← KüçükHarfeCevir(kisi1.soyisim)
                soyisim2 ← KüçükHarfeCevir(kisi2.soyisim)

                Eğer soyisim1 > soyisim2 ise
                    temp ← Rehber[j]
                    Rehber[j] ← Rehber[j+1]
                    Rehber[j+1] ← temp
                SonEğer
            SonEğer

            j ← j + 1
        SonTekrar

        i ← i + 1
    SonTekrar

    // 7. Sıralama tamamlandıktan sonra rehberi ekrana yazdır
    Yaz("Sıralanmış Telefon Rehberi:")
    k ← 0
    Tekrar k < n olduğu sürece
        Yaz("İsim: ", Rehber[k].isim, " Soyisim: ", Rehber[k].soyisim, " Telefon: ", Rehber[k].telefon)
        k ← k + 1
    SonTekrar

Bitir
