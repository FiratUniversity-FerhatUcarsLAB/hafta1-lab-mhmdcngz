Lütfen seçtiğiniz algoritmaya ait çözümü ve diğer isterleri aşağıya ekleyiniz:

SUDOKU ÇÖZME YAKLAŞIMI

Fonksiyon ÇözSudoku(tahta):
    Eğer BoşHucreBul(tahta) == YOK:
        // Tahta tamamen dolu, çözüm bulundu
        Döndür TRUE

    (satir, sutun) = BoşHucreBul(tahta)

    For sayı in 1..9:
        Eğer GeçerliMi(tahta, satir, sutun, sayı):
            tahta[satir][sutun] = sayı

            Eğer ÇözSudoku(tahta) == TRUE:
                Döndür TRUE // Çözüm bulundu

            // Geri al (backtrack)
            tahta[satir][sutun] = 0

    // Hiçbir sayı uygun değilse
    Döndür FALSE

Fonksiyon BoşHucreBul(tahta):
    For satir in 0..8:
        For sutun in 0..8:
            Eğer tahta[satir][sutun] == 0:
                Döndür (satir, sutun)
    Döndür YOK

Fonksiyon GeçerliMi(tahta, satir, sutun, sayı):
    // Satırda aynı sayı var mı?
    For x in 0..8:
        Eğer tahta[satir][x] == sayı:
            Döndür FALSE

    // Sütunda aynı sayı var mı?
    For y in 0..8:
        Eğer tahta[y][sutun] == sayı:
            Döndür FALSE

    // 3x3'lük kutuda aynı sayı var mı?
    baslangicSatir = (satir // 3) * 3
    baslangicSutun = (sutun // 3) * 3
    For i in 0..2:
        For j in 0..2:
            Eğer tahta[baslangicSatir + i][baslangicSutun + j] == sayı:
                Döndür FALSE

    Döndür TRUE
